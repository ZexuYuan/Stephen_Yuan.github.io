  ny_noaa %>%
  separate(date, c("year", "month", "day"), sep = "-") %>%

    month = as.numeric(month),
    day = as.numeric(day),
    prcp = as.numeric(prcp)*0.1,
    tmax = as.numeric(tmax)*0.1,

  drop_na(tmin, tmax)  %>%
